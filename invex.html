#include <iostream>
#include <iomanip>
#include <vector>
#include <string>
using namespace std;

struct Patrocinador {
    string nombre;
    double monto;
};

struct TipoBoleto {
    string nombre;
    double precio;
    int cantidad;
};

void mostrarEncabezado() {
    cout << "==================================================\n";
    cout << "=== ANALIZADOR DE RENTABILIDAD - WANZ FEST 2024 ===\n";
    cout << "==================================================\n\n";
}

double calcularIngresos(const vector<TipoBoleto>& boletos, double patrocinios) {
    double total = patrocinios;
    for (const auto& boleto : boletos) {
        total += boleto.precio * boleto.cantidad;
    }
    return total;
}

int calcularAsistentes(const vector<TipoBoleto>& boletos) {
    int total = 0;
    for (const auto& boleto : boletos) {
        total += boleto.cantidad;
    }
    return total;
}

double calcularCostos(double costosFijos, double costoPorPersona, int asistentes) {
    return costosFijos + (costoPorPersona * asistentes);
}

int main() {
    // Configuración de decimales
    cout << fixed << setprecision(2);
    
    // Variables principales
    vector<TipoBoleto> boletos;
    vector<Patrocinador> patrocinadores;
    double costosFijos, costoPorPersona;
    double totalPatrocinios = 0;
    
    mostrarEncabezado();
    
    // Entrada de datos básicos
    cout << "DATOS BÁSICOS DEL EVENTO:\n";
    cout << "Costos fijos (lugar, permisos, seguridad): $";
    cin >> costosFijos;
    
    cout << "Costo por persona (comida, servicios): $";
    cin >> costoPorPersona;
    
    // Tipos de boletos
    int numTipos;
    cout << "\nTIPOS DE BOLETOS DISPONIBLES:\n";
    cout << "Cuantos tipos de boletos diferentes? ";
    cin >> numTipos;
    
    for (int i = 0; i < numTipos; i++) {
        TipoBoleto boleto;
        cout << "\nBoleto #" << i+1 << ":\n";
        cout << "  Nombre (ej. General, VIP, Niño): ";
        cin.ignore();
        getline(cin, boleto.nombre);
        cout << "  Precio: $";
        cin >> boleto.precio;
        cout << "  Cantidad vendida: ";
        cin >> boleto.cantidad;
        boletos.push_back(boleto);
    }
    
    // Patrocinadores
    int numPatrocinadores;
    cout << "\nPATROCINADORES:\n";
    cout << "Cuantos patrocinadores? ";
    cin >> numPatrocinadores;
    
    for (int i = 0; i < numPatrocinadores; i++) {
        Patrocinador patrocinador;
        cout << "\nPatrocinador #" << i+1 << ":\n";
        cout << "  Nombre: ";
        cin.ignore();
        getline(cin, patrocinador.nombre);
        cout << "  Monto aportado: $";
        cin >> patrocinador.monto;
        patrocinadores.push_back(patrocinador);
        totalPatrocinios += patrocinador.monto;
    }
    
    // Cálculos
    int asistentes = calcularAsistentes(boletos);
    double ingresos = calcularIngresos(boletos, totalPatrocinios);
    double costos = calcularCostos(costosFijos, costoPorPersona, asistentes);
    double ganancia = ingresos - costos;
    double roi = (ganancia / costosFijos) * 100;
    
    // Resultados
    system("cls || clear");
    mostrarEncabezado();
    
    cout << "DETALLE DE VENTAS:\n";
    cout << "--------------------------------------------------\n";
    cout << left << setw(15) << "TIPO BOLETO" 
         << setw(10) << "PRECIO" 
         << setw(10) << "CANTIDAD" 
         << setw(15) << "TOTAL" << endl;
    
    for (const auto& boleto : boletos) {
        cout << setw(15) << boleto.nombre 
             << "$" << setw(9) << boleto.precio 
             << setw(10) << boleto.cantidad 
             << "$" << setw(14) << (boleto.precio * boleto.cantidad) << endl;
    }
    
    cout << "\nPATROCINADORES (" << patrocinadores.size() << "):\n";
    for (const auto& patrocinador : patrocinadores) {
        cout << " - " << patrocinador.nombre << ": $" << patrocinador.monto << endl;
    }
    
    cout << "\nRESUMEN FINANCIERO:\n";
    cout << "--------------------------------------------------\n";
    cout << setw(30) << "Total asistentes:" << asistentes << endl;
    cout << setw(30) << "Ingresos por boletos:" << "$" << (ingresos - totalPatrocinios) << endl;
    cout << setw(30) << "Ingresos por patrocinios:" << "$" << totalPatrocinios << endl;
    cout << setw(30) << "Total ingresos:" << "$" << ingresos << endl;
    cout << setw(30) << "Costos fijos:" << "$" << costosFijos << endl;
    cout << setw(30) << "Costos variables:" << "$" << (costos - costosFijos) << endl;
    cout << setw(30) << "Total costos:" << "$" << costos << endl;
    cout << setw(30) << "Ganancia neta:" << "$" << ganancia << endl;
    cout << setw(30) << "ROI (Retorno sobre inversión):" << roi << "%" << endl;
    
    cout << "\nRECOMENDACIÓN:\n";
    cout << "--------------------------------------------------\n";
    if (ganancia > 0) {
        if (roi > 30) {
            cout << "🌟 INVERSIÓN ALTAMENTE RECOMENDABLE (ROI excelente)\n";
        } else if (roi > 15) {
            cout << "👍 INVERSIÓN RECOMENDABLE (Buena rentabilidad)\n";
        } else {
            cout << "⚠️ INVERSIÓN ACEPTABLE (ROI moderado)\n";
        }
    } else {
        cout << "❌ INVERSIÓN NO RECOMENDABLE (Pérdida proyectada)\n";
    }
    
    cout << "\n==================================================\n";
    return 0;
}

def analizador_festivales():
    print("=== ANALIZADOR DE VIABILIDAD PARA FESTIVALES ===")
    print("===== (Modelo WANZ Fest - We Are Not Zombies) =====")
    print()
    
    # Datos de entrada
    capital_inicial = float(input("Capital inicial necesario: $"))
    costos_fijos = float(input("Costos fijos (lugar, permisos, seguridad): $"))
    costo_por_persona = float(input("Costo por persona (logística, servicios): $"))
    precio_entrada = float(input("Precio promedio por entrada: $"))
    asistentes_esperados = int(input("Asistentes estimados: "))
    
    # Ingresos por patrocinios
    patrocinadores = int(input("\nNúmero de patrocinadores: "))
    ingresos_patrocinios = 0
    for i in range(patrocinadores):
        nombre = input(f"Nombre patrocinador {i+1}: ")
        monto = float(input(f"Monto aportado por {nombre}: $"))
        ingresos_patrocinios += monto
    
    # Cálculos financieros
    ingresos_entradas = asistentes_esperados * precio_entrada
    ingresos_totales = ingresos_entradas + ingresos_patrocinios
    costos_totales = costos_fijos + (costo_por_persona * asistentes_esperados)
    ganancia_neta = ingresos_totales - costos_totales
    roi = (ganancia_neta / capital_inicial) * 100
    
    # Resultados
    print("\n=== RESULTADOS ===")
    print(f"Ingresos por entradas: ${ingresos_entradas:,.2f}")
    print(f"Ingresos por patrocinios: ${ingresos_patrocinios:,.2f}")
    print(f"Ingresos totales: ${ingresos_totales:,.2f}")
    print(f"Costos totales: ${costos_totales:,.2f}")
    print(f"Ganancia neta: ${ganancia_neta:,.2f}")
    
    # Recomendación
    print("\n=== RECOMENDACIÓN ===")
    if ganancia_neta > 0:
        print(f"✅ ROI (Retorno sobre inversión): {roi:.2f}%")
        if roi > 30:
            print("🌟 INVERSIÓN ALTAMENTE RECOMENDABLE")
        elif roi > 15:
            print("👍 INVERSIÓN RECOMENDABLE")
        else:
            print("⚠️ INVERSIÓN DE RIESGO MODERADO")
    else:
        print(f"❌ Pérdida estimada: ${abs(ganancia_neta):,.2f}")
        print("🚫 INVERSIÓN NO RECOMENDABLE")

# Ejecutar el programa
if __name__ == "__main__":
    analizador_festivales()

program GananciaPorBoleto;

uses
  SysUtils;

var
  PrecioBoleto: Real;
  CostoVariablePorPersona: Real;
  GananciaNetaPorBoleto: Real;

begin
  // Entradas
  Write('Precio del boleto: $');
  Readln(PrecioBoleto);

  Write('Costo variable por asistente (servicios, logística, etc.): $');
  Readln(CostoVariablePorPersona);

  // Cálculo
  GananciaNetaPorBoleto := PrecioBoleto - CostoVariablePorPersona;

  // Salida
  Writeln;
  Writeln('Ganancia neta por cada boleto vendido: $', FormatFloat('0.00', GananciaNetaPorBoleto));

  if GananciaNetaPorBoleto > 0 then
    Writeln('¡Felicidades! Estás obteniendo utilidad por boleto 🟢')
  else
    Writeln('Cuidado: estás perdiendo dinero por boleto 🔴');
end.
